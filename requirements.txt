# E-Commerce Web Application Frontend Requirements
# ==================================================
# This file documents the required technologies and libraries for the frontend

#
# Note: This is for documentation purposes only. Actual dependencies are managed via package.json.

# 1. Runtime Environment
# ----------------------
# Node.js is required to run and build the application
nodejs>=18.0.0

# Package managers (choose one):
# npm (comes with Node.js)
# pnpm>=8.0.0
# bun>=1.0.0

# --------------------------
# React - JavaScript library for building user interfaces
react>=18.0.0

# React DOM - DOM-specific methods for React
react-dom>=18.0.0

# 3. Build and Tooling
# --------------------
# Vite - Next generation frontend tooling
vite>=4.0.0

# TypeScript - Typed superset of JavaScript
typescript>=5.0.0

# -----------------

autoprefixer>=10.0.0

react-router-dom>=6.0.0

# React Context API - Built-in React solution for state management
# Conceptual requirement - Part of React core

# Includes:
# - Firebase Authentication

# - Firebase Hosting ready configuration

# ---------------------
# Dependencies are managed via package.json
# Install dependencies with:
# npm install
# OR
# pnpm install
# OR
# bun install

# ---------------------
